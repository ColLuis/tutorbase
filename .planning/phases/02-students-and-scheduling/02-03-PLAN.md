**StudentForm.tsx** — `src/components/students/StudentForm.tsx`

'use client' component using react-hook-form + zodResolver. Fields (in order):
1. name (required, text input)
2. parent_name (optional, text input)
3. parent_email (optional, email input)
4. parent_phone (optional, tel input)
5. subject (optional, text input — e.g. "Mathematics", "English")
6. default_rate (optional, number input with min=0, step=0.01, placeholder="75.00")
7. default_duration_minutes (optional, Select component — options: 30, 45, 60, 75, 90, 120 minutes)
8. notes (optional, Textarea)

Props interface:
```typescript
interface StudentFormProps {
  mode: 'create' | 'edit'
  studentId?: string  // only in edit mode
  defaultValues?: Partial<FormValues>
}
```

On submit in create mode: call `createStudent(formData)` — build FormData from form values.
On submit in edit mode: call `updateStudent(studentId, formData)`.
On success: use `router.push('/students')` (useRouter from next/navigation).
On error: display error message below the submit button.

Zod schema (use zod v4 method syntax):
```typescript
const StudentSchema = z.object({
  name: z.string().min(1, 'Name is required'),
  parent_name: z.string().optional(),
  parent_email: z.string().email().optional().or(z.literal('')),
  parent_phone: z.string().optional(),
  subject: z.string().optional(),
  default_rate: z.coerce.number().min(0).optional().or(z.literal('')),
  default_duration_minutes: z.coerce.number().int().min(15).optional().or(z.literal('')),
  notes: z.string().optional(),
})
```

Submit button: min-h-[44px], full width on mobile.

**src/app/(app)/students/[id]/page.tsx**
```typescript
import { notFound } from 'next/navigation'
import Link from 'next/link'
import { verifySession } from '@/lib/dal'
import { getStudent } from '@/lib/queries/students'
import DeactivateButton from '@/components/students/DeactivateButton'
import { Button } from '@/components/ui/button'
import { Badge } from '@/components/ui/badge'
import { formatCurrency } from '@/lib/utils/currency'

export default async function StudentDetailPage({ params }: { params: Promise<{ id: string }> }) {
  const { tutorId } = await verifySession()
  const { id } = await params
  let student
  try { student = await getStudent(tutorId, id) } catch { notFound() }

  return (
    <div className="p-4 md:p-6 max-w-lg mx-auto">
      <div className="flex items-center gap-4 mb-6">
        <Link href="/students" className="text-muted-foreground text-sm">← Students</Link>
      </div>
      <div className="flex items-start justify-between mb-4">
        <h1 className="text-2xl font-bold">{student.name}</h1>
        {!student.is_active && <Badge variant="secondary">Inactive</Badge>}
      </div>
      <div className="space-y-3 mb-8">
        {student.subject && <p><span className="font-medium">Subject:</span> {student.subject}</p>}
        {student.parent_name && <p><span className="font-medium">Parent:</span> {student.parent_name}</p>}
        {student.parent_email && <p><span className="font-medium">Email:</span> {student.parent_email}</p>}
        {student.parent_phone && <p><span className="font-medium">Phone:</span> {student.parent_phone}</p>}
        {student.default_rate != null && <p><span className="font-medium">Rate:</span> {formatCurrency(student.default_rate)} / hr</p>}
        {student.default_duration_minutes != null && <p><span className="font-medium">Duration:</span> {student.default_duration_minutes} min</p>}
        {student.notes && <p><span className="font-medium">Notes:</span> {student.notes}</p>}
      </div>
      <div className="flex gap-3">
        <Button asChild variant="outline">
          <Link href={`/students/${id}/edit`}>Edit</Link>
        </Button>
        {student.is_active && <DeactivateButton studentId={id} studentName={student.name} />}
      </div>
    </div>
  )
}
```
